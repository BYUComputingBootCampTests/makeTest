TARGET=tree
CC=g++
HEADERS=$(wildcard *.h)
SRCS=$(wildcard *.cpp)
OBJS=$(SRCS:.cpp=.o)

$(TARGET): $(OBJS)
	$(CC) $(OBJS) -o $(TARGET)
	touch compiled.txt
	@echo "$(HEADERS)" >> compiled.txt
	@echo "$(SRCS)" >> compiled.txt
    

%.o: %.cpp
	$(CC) -c $^

clean:
	rm $(TARGET) $(OBJS)
	rm compiled.txt
