Target=tree
Sources=$(wildcard *.cpp)
Objects=$(Sources:.cpp=.o)
Headers=$(wildcard *.h)

$(Target): $(Objects) $(Headers)
	g++ $(Objects) -o $(Target)
	ls $(Objects) > compiled.txt 
	ls $(Sources) >> compiled.txt
	ls $(Headers) >> compiled.txt 

%.o: %.c $(Headers)
	g++ $< -c 

clean: 
	rm tree
	rm *.o
	rm compiled.txt
